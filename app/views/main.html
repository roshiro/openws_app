The REST API for your JSON.

After you enter your email address, you'll receive the information to use REST API straight away in your application.

Get it working within 5 minutes!

Follow the steps to use the API with AngularJS.

#1. Saving a JSON object

    angular.module('openws')
    .controller('ExampleController', ['$http', $log, function ($http, $log,$scope) {

      var = taskJSON = {
        name: "My first task",
        completed: false
      };

      $http.post("https://openws.herokuapp.com/my_tasks", taskJSON)
      .success(function() {
        $log.debug("Task saved successfuly");
      })
      .error(function() {
        $log.error("Task could not be saved.");
      });

    }
